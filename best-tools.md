// ItemList Schema (generated dynamically)
(function() {
  const items = [];
  let position = 1;
  document.querySelectorAll('.tool-card').forEach(card => {
    const nameEl = card.querySelector('.tool-name');
    const linkEl = card.querySelector('.card-cta a:first-child');
    const reviewLinkEl = card.querySelector('.review-link');
    if (nameEl && linkEl) {
      const name = nameEl.innerText.trim();
      const url = linkEl.href;
      const reviewUrl = reviewLinkEl ? reviewLinkEl.href : null;
      items.push({
        "@type": "ListItem",
        "position": position,
        "item": {
          "@type": "Product",
          "name": name,
          "url": url,
          "review": reviewUrl ? { "@type": "Review", "url": reviewUrl } : undefined
        }
      });
      position++;
    }
  });
  const itemListSchema = {
    "@context": "https://schema.org",
    "@type": "ItemList",
    "name": "Best Small Business Tools 2026",
    "description": "A curated list of recommended software tools for small businesses, organized by category.",
    "numberOfItems": items.length,
    "itemListElement": items
  };
  const script = document.createElement('script');
  script.type = 'application/ld+json';
  script.textContent = JSON.stringify(itemListSchema);
  document.head.appendChild(script);
})();
