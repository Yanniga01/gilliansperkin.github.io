<script>
  function acceptAllCookies() {
    localStorage.setItem('cookieConsent', 'all');
    localStorage.setItem('analyticsCookies', 'true');
    localStorage.setItem('marketingCookies', 'true');
    alert('All cookies accepted. Thank you!');
  }
  function rejectNonEssential() {
    localStorage.setItem('cookieConsent', 'essential');
    localStorage.setItem('analyticsCookies', 'false');
    localStorage.setItem('marketingCookies', 'false');
    alert('Only essential cookies will be used.');
  }
  function savePreferences() {
    const analytics = document.getElementById('analyticsCookies').checked;
    const marketing = document.getElementById('marketingCookies').checked;
    localStorage.setItem('cookieConsent', 'custom');
    localStorage.setItem('analyticsCookies', analytics);
    localStorage.setItem('marketingCookies', marketing);
    alert('Cookie preferences saved!');
  }
  document.addEventListener('DOMContentLoaded', function() {
    const analyticsPref = localStorage.getItem('analyticsCookies');
    const marketingPref = localStorage.getItem('marketingCookies');
    if (analyticsPref !== null) {
      const analyticsCheck = document.getElementById('analyticsCookies');
      if (analyticsCheck) analyticsCheck.checked = analyticsPref === 'true';
    }
    if (marketingPref !== null) {
      const marketingCheck = document.getElementById('marketingCookies');
      if (marketingCheck) marketingCheck.checked = marketingPref === 'true';
    }
    // Optional: hide a banner if you have one – not included here as layout handles it
  });
</script>
